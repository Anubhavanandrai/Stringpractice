//palindrome

import java.util.*;

class Palindrome{
    
    public boolean CheckPalindrome( String s){
        
        int i=0;
        int j= s.length()-1;
        
        while(i<=j)
        {
            if(s.charAt(i)!=s.charAt(j))
            {
                return false;
            }
            j--;
            i++;
        }
       return true; 
    }
    
}


public class Main
{
	public static void main(String[] args) {
	    
	    String s= "ASBjjBSA";
	    Palindrome obj = new Palindrome();
	    boolean answer = obj.CheckPalindrome(s);
		System.out.println(answer);
	}
}
